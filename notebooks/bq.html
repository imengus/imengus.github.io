def upload_csv(path, table_id, client=client, job_config=bigquery.LoadJobConfig(
    autodetect=True,  # Infer schema from CSV.
    skip_leading_rows=1,  # Skip header row.
    max_bad_records=10,  # Tolerate up to 10 bad records.
    source_format=bigquery.SourceFormat.CSV,  # Source format is CSV.
)):
    with open(path, "rb") as source_file:
        job = client.load_table_from_file(source_file, table_id, job_config=job_config)
    job.result()  # Wait for job completion.
    destination_table = client.get_table(table_id)
    print("Loaded {} rows.".format(destination_table.num_rows))

def upload_df(df, table_id, client=client, job_config=bigquery.LoadJobConfig(
    autodetect=True,  # Infer schema from DataFrame.
    max_bad_records=10,  # Tolerate up to 10 bad records.
    write_disposition="WRITE_TRUNCATE"  # Overwrite existing table.
)):
    job = client.load_table_from_dataframe(df, table_id, job_config=job_config)
    job.result()  # Wait for job completion.
    destination_table = client.get_table(table_id)
    print("Loaded {} rows.".format(destination_table.num_rows))

def delete_other_tables(dataset_id, tab_name, client=client):
    get_tabs = lambda dataset_id: [i.table_id for i in client.list_tables(dataset_id)]
    table_names = get_tabs(dataset_id)

    for i in table_names:
        if i == tab_name:
            continue
        client.delete_table(f"{dataset_id}.{i}", not_found_ok=True)

def query_df(QUERY, client=client):
    return client.query_and_wait(QUERY).to_dataframe()

def table_summary(frame):
    des = frame.describe().transpose()
    summary = frame.nunique(0).to_frame("n unique values")
    summary['n zeros'] = len(frame) - frame.astype(bool).sum(axis=0)
    summary['median'] = des['50%'].astype(int)
    summary['mean'] = des['mean'].astype(int)
    summary['max'] = des['max'].astype(int)
    return summary.fillna(0)

def corr_plot(frame, xtick_labels, method="spearman"):

    fig, ax = plt.subplots(1, 2)
    fig.set_figwidth(10)
    
    numerics = frame[frame.columns[(frame.dtypes == pd.Int64Dtype()) | (frame.dtypes == float)]]
    mat = numerics.corr(method).to_numpy()

    sns.heatmap(mat, cbar=False, annot=True, square=False, fmt='.2f', yticklabels=xtick_labels, xticklabels=xtick_labels, cmap="Spectral_r", ax=ax[0])
    ax[0].set_title(f"{method.upper()} correlation coefficients")

    dissimilarity = 1 - abs(mat)

    # Perform hierarchical clustering and plot dendrogram.
    Z = linkage(squareform(dissimilarity))

    dendrogram(Z, labels=xtick_labels, orientation='left', 
            leaf_rotation=0,
            color_threshold=0.2,
            ax=ax[1])
    ax[1].set_title('Hierarchical similarity dendrogram')
    ax[1].set_xlabel("Distance")
    ax[1].spines['left'].set_visible(False)
    ax[1].spines['right'].set_visible(True)
    ax[1].set_xlim(1, 0)
    plt.show()

### Plot medians with error bars for each agency

medians = []
errors = []
for col in cpi:

    # Select non-null values from the column. ',' to make it a tuple
    s = cpi[col]
    s = s[s.notna()],

    # Compute confidence interval for median using bootstrap resampling
    l, u = stats.bootstrap(s, lambda s: np.median(s)).confidence_interval
    medians.append(s[0].median())
    errors.append((l, u))

preerr = np.array(errors).T.tolist()
meds = np.array([medians]).repeat(2, axis=0)
err = np.abs(preerr - meds)

plt.bar(cpi.columns, medians, alpha=0.5)
plt.errorbar(np.arange(6), medians, yerr=err, fmt='_', zorder=100, lw=3, gapcolor='k', mec='k', color='0.4')
plt.ylim(0, 20)
plt.xlim(-0.5, 5.5)
plt.yticks([i for i in range(4, 20, 4)])
plt.title("Medians of Cost / 1000 Imps, with error bars")
plt.show()
  • The bootstrapped 95% confidence intervals of the medians of the top 2 do not overlap with others
  • angora is the most cost effective solution at with median Cost per 1000 Impressions at 1.86 (+- 0.14), followed by apostates at 2.8 (+0.3, -0.14), with altercations and apostates tied at third
  • Means are different from medians: 1.9 != 2.5 and 2.8 != 3.0. This is because of large outliers that we could see in the opacity comparison graph
  • baleen and amour have very unreliable cost efficiency estimates, so we are only picking among the top 4
